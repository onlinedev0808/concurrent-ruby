def join(*tokens, &block)
  block ||= -> tokens { Promises.any_event(*tokens.map(&:to_event)) }
  self.class.new block.call([@Cancel, *tokens])
end

def loop_until_canceled(&block)
  until canceled?
    result = block.call
  end
  result
end
