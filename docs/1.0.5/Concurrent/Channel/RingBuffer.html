def offer(value)
  return false if full?

  @buffer[@last] = value
  @last = (@last + 1) % @buffer.size
  @count += 1
  true
end

def poll
  result = @buffer[@first]
  @buffer[@first] = nil
  @first = (@first + 1) % @buffer.size
  @count -= 1
  result
end
