def on_envelope(envelope)
  case envelope.message
    # only from supervisor?
  when :reset!
  when :restart!
  when :continue!
  else
    if @paused
      @buffer << envelope
    else
      begin
        pass envelope
      rescue => error
        #
      end
    end
  end
end

def reject_messages
  @buffer.each { |envelope| reject_envelope envelope }
  @buffer.clear
  super
end
