def kill
  mutex.synchronize do
    break if shutdown?
    stop_event.set
    kill_execution
    stopped_event.set
  end
  true
end

def post(*args, &task)
  raise ArgumentError.new('no block given') unless block_given?
  mutex.synchronize do
    # If the executor is shut down, reject this task
    return handle_fallback(*args, &task) unless running?
    execute(*args, &task)
    true
  end
end

  end
