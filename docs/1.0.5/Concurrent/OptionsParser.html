def get_executor_from(opts = {})
  if opts[:executor]
    opts[:executor]
  elsif opts[:operation] == true || opts[:task] == false
    Concurrent.configuration.global_operation_pool
  elsif opts[:operation] == false || opts[:task] == true
    Concurrent.configuration.global_task_pool
  else
    nil
  end
end
